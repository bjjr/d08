- The lessor/s whose ratio of requested versus approved request/s is the maximum or the minimum. (Ratios must be rounded to one decimal place)

FORMA 1: Ratio de todas las solicitudes (Aceptadas, Denegadas y Pendientes) con respecto a las Aceptadas.

select l from Lessor l join l.properties p order by (p.books.size/(select count(b) from Lessor l2 join l2.properties p join p.books b where b.status.name = 'ACCEPTED')) asc;
select l from Lessor l join l.properties p order by (p.books.size/(select count(b) from Lessor l2 join l2.properties p join p.books b where b.status.name = 'ACCEPTED')) desc;

FORMA 2: Ratio de todas las solicitudes Pendientes con respecto a las Aceptadas.

select l from Lessor l join l.properties p order by ((select count(b1) from Lessor l1 join l1.properties p1 join p1.books b1 where b1.status.name = 'PENDING')/(select count(b2) from Lessor l2 join l2.properties p join p.books b2 where b2.status.name = 'ACCEPTED')) asc;
select l from Lessor l join l.properties p order by ((select count(b1) from Lessor l1 join l1.properties p1 join p1.books b1 where b1.status.name = 'PENDING')/(select count(b2) from Lessor l2 join l2.properties p join p.books b2 where b2.status.name = 'ACCEPTED')) desc;


- The tenant/s whose ratio of requested versus approved request/s is the maximum or the minimum. (Ratios must be rounded to one decimal place)

FORMA 1: Ratio de todas las solicitudes (Aceptadas, Denegadas y Pendientes) con respecto a las Aceptadas.

select t from Tenant t order by (t.books.size/(select count(b) from Tenant t2 join t2.books b where b.status.name = 'ACCEPTED')) asc;
select t from Tenant t order by (t.books.size/(select count(b) from Tenant t2 join t2.books b where b.status.name = 'ACCEPTED')) desc;

FORMA 2: Ratio de todas las solicitudes Pendientes con respecto a las Aceptadas.

select t from Tenant t order by ((select count(b1) from Tenant t1 join t1.books b1 where b1.status.name = 'PENDING')/(select count(b2) from Tenant t2 join t2.books b2 where b2.status.name = 'ACCEPTED')) asc;

- The average, the minimum, and the maximum number of results per finder

select avg(b.property), max(b.property), min(b.property) from Tenant t left join t.books b where t.finder.keyword like b.property.name or t.finder.keyword like b.property.description or t.finder.keyword like b.property.address;
